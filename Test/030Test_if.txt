begin
	a:=2 + (2-3);

	c:=3 + (2-3) * (2*3);
	d:=-2 + 2 * 3 ;
	if a>b then begin
		c:=(a+b)*(a-b)/(a+b)-(a-b); 
		d:=(a-b)+(a-b)*(a+b); 
		e:= c -d; 
	end else begin
		c:=(a+b)*(a-b)/(a+b)-(a-b); 
		d:=(a-b)+(a-b)*(a+b);
